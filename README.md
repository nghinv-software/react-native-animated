## Installation

```sh
yarn add @nghinv/react-native-animated
```

or 

```sh
npm install @nghinv/react-native-animated
```

## Usage

```js
import { 
  springConfig, 
  timingConfig,
  colors,
  useHsv,
  useRgb,
  useVector,
  useLayoutRectangle,
  toDeg,
  toRadian,
  clamp,
  mix,
  convertUtf8,
  canvas2Polar,
  polar2Canvas,
  vec,
  vecAnimated,
} from '@nghinv/react-native-animated';
```
